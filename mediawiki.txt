sudo apt install --no-install-recommends imagemagick mediawiki mediawiki-extensions-math mediawiki-math mysql-server php5-gd

wget https://dl.eff.org/certbot-auto
sudo mv certbot-auto /usr/local/sbin/certbot-auto
chmod 755 /usr/local/sbin/certbot-auto
certbot-auto --apache -d ywsing.onedse.com

echo '30 2 * * 1 /usr/local/sbin/certbot-auto renew >> /var/log/le-renew.log' | sudo tee /etc/cron.weekly/certbot-auto
sudo chmod 755 /etc/cron.weekly/certbot-auto

sudo a2enconf mediawiki
sudo sed 's/#Alias/Alias/g' -i /etc/mediawiki/apache.conf
sudo service apache2 reload
sudo /etc/init.d/apache2 restart

firefox https://ywsing.onedse.com/mediawiki/
